### 💻 本地部署实现

```python
import torch
from transformers import AutoModelForCausalLM, AutoTokenizer
from flask import Flask, request, jsonify
import threading
import queue
import time

class LocalLLMService:
    def __init__(self, model_path, device="cuda"):
        self.device = device
        self.model = AutoModelForCausalLM.from_pretrained(
            model_path,
            torch_dtype=torch.float16,
            device_map="auto"
        )
        self.tokenizer = AutoTokenizer.from_pretrained(model_path)
        
        self.request_queue = queue.Queue()
        self.response_queues = {}
        
        self.worker_thread = threading.Thread(target=self._process_requests)
        self.worker_thread.daemon = True
        self.worker_thread.start()
    
    def _process_requests(self):
        while True:
            request_id, prompt, params = self.request_queue.get()
            
            try:
                response = self._generate(prompt, params)
                self.response_queues[request_id].put({
                    'status': 'success',
                    'response': response
                })
            except Exception as e:
                self.response_queues[request_id].put({
                    'status': 'error',
                    'error': str(e)
                })
    
    def _generate(self, prompt, params):
        inputs = self.tokenizer(prompt, return_tensors="pt").to(self.device)
        
        with torch.no_grad():
            outputs = self.model.generate(
                inputs.input_ids,
                max_length=params.get('max_length', 100),
                temperature=params.get('temperature', 0.7),
                top_p=params.get('top_p', 0.9),
                do_sample=params.get('do_sample', True)
            )
        
        response = self.tokenizer.decode(outputs[0], skip_special_tokens=True)
        
        return response
    
    def generate_async(self, prompt, params):
        request_id = str(time.time())
        
        self.response_queues[request_id] = queue.Queue()
        
        self.request_queue.put((request_id, prompt, params))
        
        return request_id
    
    def get_response(self, request_id, timeout=30):
        try:
            response = self.response_queues[request_id].get(timeout=timeout)
            del self.response_queues[request_id]
            return response
        except queue.Empty:
            return {'status': 'timeout', 'error': 'Request timeout'}

app = Flask(__name__)

llm_service = None

@app.route('/init', methods=['POST'])
def init_service():
    global llm_service
    
    data = request.json
    model_path = data.get('model_path')
    device = data.get('device', 'cuda')
    
    llm_service = LocalLLMService(model_path, device)
    
    return jsonify({'status': 'initialized'})

@app.route('/generate', methods=['POST'])
def generate():
    if llm_service is None:
        return jsonify({'status': 'error', 'error': 'Service not initialized'})
    
    data = request.json
    prompt = data.get('prompt')
    params = data.get('params', {})
    
    request_id = llm_service.generate_async(prompt, params)
    
    response = llm_service.get_response(request_id)
    
    return jsonify(response)

@app.route('/generate_async', methods=['POST'])
def generate_async():
    if llm_service is None:
        return jsonify({'status': 'error', 'error': 'Service not initialized'})
    
    data = request.json
    prompt = data.get('prompt')
    params = data.get('params', {})
    
    request_id = llm_service.generate_async(prompt, params)
    
    return jsonify({'status': 'queued', 'request_id': request_id})

@app.route('/get_response/<request_id>', methods=['GET'])
def get_response(request_id):
    if llm_service is None:
        return jsonify({'status': 'error', 'error': 'Service not initialized'})
    
    response = llm_service.get_response(request_id)
    
    return jsonify(response)

def run_local_service():
    app.run(host='0.0.0.0', port=5000, threaded=True)

if __name__ == "__main__":
    run_local_service()
```

```python
import docker
import yaml
import subprocess
import json

class CloudDeployment:
    def __init__(self, cloud_provider="aws"):
        self.cloud_provider = cloud_provider
        self.docker_client = docker.from_env()
    
    def build_docker_image(self, model_path, image_name):
        dockerfile = """
FROM python:3.9

WORKDIR /app

COPY requirements.txt .
RUN pip install -r requirements.txt

COPY . .

CMD ["python", "server.py"]
"""
        
        with open("Dockerfile", "w") as f:
            f.write(dockerfile)
        
        image, build_logs = self.docker_client.images.build(
            path=".",
            tag=image_name,
            rm=True
        )
        
        return image
    
    def push_image(self, image_name, registry):
        image = self.docker_client.images.get(image_name)
        
        image.tag(registry, tag="latest")
        
        push_logs = self.docker_client.images.push(registry, tag="latest")
        
        return push_logs
    
    def create_kubernetes_deployment(self, deployment_config):
        deployment_yaml = {
            'apiVersion': 'apps/v1',
            'kind': 'Deployment',
            'metadata': {
                'name': deployment_config['name'],
                'labels': {
                    'app': deployment_config['name']
                }
            },
            'spec': {
                'replicas': deployment_config['replicas'],
                'selector': {
                    'matchLabels': {
                        'app': deployment_config['name']
                    }
                },
                'template': {
                    'metadata': {
                        'labels': {
                            'app': deployment_config['name']
                        }
                    },
                    'spec': {
                        'containers': [{
                            'name': deployment_config['name'],
                            'image': deployment_config['image'],
                            'ports': [{
                                'containerPort': deployment_config['port']
                            }],
                            'resources': {
                                'limits': {
                                    'nvidia.com/gpu': deployment_config['gpu_count']
                                }
                            }
                        }]
                    }
                }
            }
        }
        
        with open("deployment.yaml", "w") as f:
            yaml.dump(deployment_yaml, f)
        
        subprocess.run(["kubectl", "apply", "-f", "deployment.yaml"])
        
        return deployment_yaml
    
    def create_kubernetes_service(self, service_config):
        service_yaml = {
            'apiVersion': 'v1',
            'kind': 'Service',
            'metadata': {
                'name': service_config['name']
            },
            'spec': {
                'type': 'LoadBalancer',
                'selector': {
                    'app': service_config['app_name']
                },
                'ports': [{
                    'protocol': 'TCP',
                    'port': service_config['port'],
                    'targetPort': service_config['target_port']
                }]
            }
        }
        
        with open("service.yaml", "w") as f:
            yaml.dump(service_yaml, f)
        
        subprocess.run(["kubectl", "apply", "-f", "service.yaml"])
        
        return service_yaml
    
    def setup_autoscaling(self, autoscaling_config):
        hpa_yaml = {
            'apiVersion': 'autoscaling/v2',
            'kind': 'HorizontalPodAutoscaler',
            'metadata': {
                'name': autoscaling_config['name']
            },
            'spec': {
                'scaleTargetRef': {
                    'apiVersion': 'apps/v1',
                    'kind': 'Deployment',
                    'name': autoscaling_config['deployment_name']
                },
                'minReplicas': autoscaling_config['min_replicas'],
                'maxReplicas': autoscaling_config['max_replicas'],
                'metrics': [{
                    'type': 'Resource',
                    'resource': {
                        'name': 'cpu',
                        'target': {
                            'type': 'Utilization',
                            'averageUtilization': autoscaling_config['cpu_target']
                        }
                    }
                }]
            }
        }
        
        with open("hpa.yaml", "w") as f:
            yaml.dump(hpa_yaml, f)
        
        subprocess.run(["kubectl", "apply", "-f", "hpa.yaml"])
        
        return hpa_yaml
    
    def deploy_to_cloud(self, config):
        image = self.build_docker_image(config['model_path'], config['image_name'])
        
        self.push_image(config['image_name'], config['registry'])
        
        deployment = self.create_kubernetes_deployment({
            'name': config['deployment_name'],
            'replicas': config['initial_replicas'],
            'image': config['registry'],
            'port': config['port'],
            'gpu_count': config['gpu_count']
        })
        
        service = self.create_kubernetes_service({
            'name': config['service_name'],
            'app_name': config['deployment_name'],
            'port': config['external_port'],
            'target_port': config['port']
        })
        
        autoscaling = self.setup_autoscaling({
            'name': config['hpa_name'],
            'deployment_name': config['deployment_name'],
            'min_replicas': config['min_replicas'],
            'max_replicas': config['max_replicas'],
            'cpu_target': config['cpu_target']
        })
        
        return {
            'deployment': deployment,
            'service': service,
            'autoscaling': autoscaling
        }
```

### 💻 推理优化实现

```python
import torch
import torch.nn as nn
from typing import Dict, List
import time

class InferenceOptimizer:
    def __init__(self, model):
        self.model = model
        self.kv_cache = {}
    
    def quantize_model(self, quantization_type="int8"):
        if quantization_type == "int8":
            quantized_model = torch.quantization.quantize_dynamic(
                self.model,
                {nn.Linear},
                dtype=torch.qint8
            )
        elif quantization_type == "int4":
            quantized_model = self.quantize_int4(self.model)
        
        return quantized_model
    
    def quantize_int4(self, model):
        for name, module in model.named_modules():
            if isinstance(module, nn.Linear):
                weight = module.weight.data
                
                scale = weight.abs().max() / 7
                
                quantized_weight = torch.clamp(
                    torch.round(weight / scale),
                    min=-8, max=7
                )
                
                module.weight.data = quantized_weight * scale
        
        return model
    
    def optimize_kv_cache(self, max_cache_size=1000):
        self.kv_cache = {
            'K': [],
            'V': [],
            'max_size': max_cache_size
        }
    
    def update_kv_cache(self, K, V):
        if len(self.kv_cache['K']) >= self.kv_cache['max_size']:
            self.kv_cache['K'] = self.kv_cache['K'][-self.kv_cache['max_size']:]
            self.kv_cache['V'] = self.kv_cache['V'][-self.kv_cache['max_size']:]
        
        self.kv_cache['K'].append(K)
        self.kv_cache['V'].append(V)
    
    def batch_inference(self, prompts: List[str], tokenizer, batch_size=8):
        all_inputs = []
        
        for prompt in prompts:
            inputs = tokenizer(prompt, return_tensors="pt")
            all_inputs.append(inputs)
        
        results = []
        
        for i in range(0, len(all_inputs), batch_size):
            batch_inputs = all_inputs[i:i+batch_size]
            
            batch_tensor = torch.cat([inp['input_ids'] for inp in batch_inputs], dim=0)
            
            with torch.no_grad():
                outputs = self.model.generate(
                    batch_tensor,
                    max_length=100,
                    temperature=0.7
                )
            
            batch_results = [
                tokenizer.decode(output, skip_special_tokens=True)
                for output in outputs
            ]
            
            results.extend(batch_results)
        
        return results
    
    def speculative_decoding(self, prompt, draft_model, target_model, tokenizer):
        inputs = tokenizer(prompt, return_tensors="pt")
        
        draft_outputs = draft_model.generate(
            inputs.input_ids,
            max_length=50,
            temperature=0.7
        )
        
        target_outputs = target_model.generate(
            draft_outputs,
            max_length=100,
            temperature=0.7
        )
        
        return tokenizer.decode(target_outputs[0], skip_special_tokens=True)
    
    def measure_performance(self, prompt, tokenizer, num_runs=10):
        inputs = tokenizer(prompt, return_tensors="pt")
        
        latencies = []
        
        for _ in range(num_runs):
            start_time = time.time()
            
            with torch.no_grad():
                outputs = self.model.generate(
                    inputs.input_ids,
                    max_length=100
                )
            
            end_time = time.time()
            
            latency = end_time - start_time
            latencies.append(latency)
        
        avg_latency = sum(latencies) / len(latencies)
        
        output_length = outputs.shape[1] - inputs.input_ids.shape[1]
        
        tokens_per_second = output_length / avg_latency
        
        return {
            'avg_latency': avg_latency,
            'tokens_per_second': tokens_per_second,
            'output_length': output_length
        }

class FlashAttentionImplementation:
    def __init__(self, d_model, n_heads):
        self.d_model = d_model
        self.n_heads = n_heads
        self.head_dim = d_model // n_heads
    
    def flash_attention(self, Q, K, V, block_size=64):
        batch_size, seq_len, _ = Q.shape
        
        output = torch.zeros_like(Q)
        
        for i in range(0, seq_len, block_size):
            Q_block = Q[:, i:i+block_size, :]
            
            block_output = torch.zeros_like(Q_block)
            block_max = torch.full((batch_size, block_size, 1), float('-inf'))
            block_sum = torch.zeros((batch_size, block_size, 1))
            
            for j in range(0, seq_len, block_size):
                K_block = K[:, j:j+block_size, :]
                V_block = V[:, j:j+block_size, :]
                
                scores = torch.matmul(Q_block, K_block.transpose(-2, -1))
                scores = scores / (self.head_dim ** 0.5)
                
                new_max = torch.maximum(block_max, scores.max(dim=-1, keepdim=True)[0])
                
                exp_scores = torch.exp(scores - new_max)
                
                correction = torch.exp(block_max - new_max)
                
                block_sum = correction * block_sum + exp_scores.sum(dim=-1, keepdim=True)
                
                block_output = correction * block_output + torch.matmul(exp_scores, V_block)
                
                block_max = new_max
            
            output[:, i:i+block_size, :] = block_output / block_sum
        
        return output

class BatchProcessor:
    def __init__(self, model, tokenizer, max_batch_size=32):

        self.tokenizer = tokenizer
```

### 🎯 模型压缩方法

```
┌─────────────────────────────────────────────────────────┐
│         模型压缩：减小模型大小和计算量                    │
└─────────────────────────────────────────────────────────┘

压缩方法：
┌─────────────────────────────────┐
│  1. 量化（Quantization）        │
│  - INT8量化                      │
│  - INT4量化                      │
│  - 混合精度量化                  │
│                                 │
│  2. 剪枝（Pruning）              │
│  - 权重剪枝                      │
│  - 结构剪枝                      │
│  - 注意力剪枝                    │
│                                 │
│  3. 蒸馏（Distillation）        │
│  - 知识蒸馏                      │
│  - 任务蒸馏                      │
│  - 特征蒸馏                      │
│                                 │
│  效果对比：                      │
│  ┌──────────┬──────────┬──────┐│
│  │ 方法     │ 压缩率   │ 精度 ││
│  ├──────────┼──────────┼──────┤│
│  │ INT8量化 │ 4倍      │ 损失1%││
│  │ INT4量化 │ 8倍      │ 损失3%││
│  │ 剪枝50%  │ 2倍      │ 损失2%││
│  │ 蒸馏     │ 可变     │ 损失小││
│  └──────────┴──────────┴──────┘│
└─────────────────────────────────┘

量化原理：
┌─────────────────────────────────┐
│  FP32 → INT8：                  │
│                                 │
│  原始权重：FP32                  │
│  范围：[-3.4e38, 3.4e38]        │
│                                 │
│  量化权重：INT8                  │
│  范围：[-128, 127]              │
│                                 │
│  量化公式：                      │
│  quantized = round(weight / scale)│
│  dequantized = quantized * scale│
│                                 │
│  scale计算：                     │
│  scale = max(|weight|) / 127    │
│                                 │
│  例如：                         │
│  weight = 0.5                   │
│  scale = 1.0 / 127 = 0.0079     │
│  quantized = round(0.5 / 0.0079) = 63│
│  dequantized = 63 * 0.0079 = 0.497│
│                                 │
│  精度损失：0.5 - 0.497 = 0.003   │
│  损失很小                        │
└─────────────────────────────────┘
```

```python
import torch
import torch.nn as nn
import numpy as np

class ModelQuantizer:
    def __init__(self, model):
        self.model = model
    
    def quantize_int8(self):
        quantized_model = torch.quantization.quantize_dynamic(
            self.model,
            {nn.Linear},
            dtype=torch.qint8
        )
        
        return quantized_model
    
    def quantize_int4(self):
        for name, module in self.model.named_modules():
            if isinstance(module, nn.Linear):
                weight = module.weight.data
                
                scale = weight.abs().max() / 7
                
                quantized_weight = torch.clamp(
                    torch.round(weight / scale),
                    min=-8, max=7
                )
                
                module.weight.data = quantized_weight * scale
                module.weight_scale = scale
        
        return self.model
    
    def mixed_precision_quantize(self, sensitivity_analysis):
        for name, module in self.model.named_modules():
            if isinstance(module, nn.Linear):
                sensitivity = sensitivity_analysis.get(name, 1.0)
                
                if sensitivity < 0.5:
                    module.weight.data = self.quantize_weights(module.weight.data, "int4")
                elif sensitivity < 0.8:
                    module.weight.data = self.quantize_weights(module.weight.data, "int8")
        
        return self.model
    
    def quantize_weights(self, weights, quantization_type):
        if quantization_type == "int8":
            scale = weights.abs().max() / 127
            quantized = torch.clamp(torch.round(weights / scale), min=-128, max=127)
            return quantized * scale
        elif quantization_type == "int4":
            scale = weights.abs().max() / 7
            quantized = torch.clamp(torch.round(weights / scale), min=-8, max=7)
            return quantized * scale
        
        return weights
    
    def analyze_sensitivity(self, validation_data):
        sensitivity = {}
        
        original_output = self.model(validation_data)
        
        for name, module in self.model.named_modules():
            if isinstance(module, nn.Linear):
                original_weight = module.weight.data.clone()
                
                module.weight.data = torch.zeros_like(module.weight.data)
                
                quantized_output = self.model(validation_data)
                
                error = torch.mean((original_output - quantized_output) ** 2).item()
                
                sensitivity[name] = error
                
                module.weight.data = original_weight
        
        return sensitivity

class ModelPruner:
    def __init__(self, model):
        self.model = model
    
    def weight_pruning(self, pruning_ratio=0.5):
        for name, module in self.model.named_modules():
            if isinstance(module, nn.Linear):
                weight = module.weight.data
                
                threshold = torch.topk(
                    weight.abs().flatten(),
                    int(weight.numel() * pruning_ratio)
                )[0].min()
                
                mask = weight.abs() > threshold
                
                module.weight.data = weight * mask
        
        return self.model
    
    def structured_pruning(self, pruning_ratio=0.5):
        for name, module in self.model.named_modules():
            if isinstance(module, nn.Linear):
                weight = module.weight.data
                
                neuron_importance = weight.abs().sum(dim=1)
                
                num_neurons = weight.shape[0]
                num_keep = int(num_neurons * (1 - pruning_ratio))
                
                keep_indices = torch.topk(neuron_importance, num_keep)[1]
                
                module.weight.data = weight[keep_indices, :]
                
                if module.bias is not None:
                    module.bias.data = module.bias.data[keep_indices]
        
        return self.model
    
    def attention_pruning(self, pruning_ratio=0.5):
        for name, module in self.model.named_modules():
            if hasattr(module, 'attention'):
                attention_weights = module.attention.weight.data
                
                head_importance = attention_weights.abs().sum(dim=[1, 2])
                
                num_heads = attention_weights.shape[0]
                num_keep = int(num_heads * (1 - pruning_ratio))
                
                keep_indices = torch.topk(head_importance, num_keep)[1]
                
                module.attention.weight.data = attention_weights[keep_indices, :, :]
        
        return self.model
    
    def fine_tune_after_pruning(self, training_data, epochs=5):
        optimizer = torch.optim.Adam(self.model.parameters(), lr=1e-5)
        
        for epoch in range(epochs):
            for batch in training_data:
                optimizer.zero_grad()
                
                output = self.model(batch['input'])
                loss = torch.nn.functional.cross_entropy(output, batch['target'])
                
                loss.backward()
                optimizer.step()
        
        return self.model

class KnowledgeDistillation:
    def __init__(self, teacher_model, student_model):
        self.teacher_model = teacher_model
        self.student_model = student_model
    
    def distill(self, training_data, epochs=10, temperature=2.0, alpha=0.5):
        optimizer = torch.optim.Adam(self.student_model.parameters(), lr=1e-4)
        
        for epoch in range(epochs):
            for batch in training_data:
                optimizer.zero_grad()
                
                with torch.no_grad():
                    teacher_output = self.teacher_model(batch['input'])
                    teacher_probs = torch.nn.functional.softmax(
                        teacher_output / temperature, dim=-1
                    )
                
                student_output = self.student_model(batch['input'])
                student_probs = torch.nn.functional.log_softmax(
                    student_output / temperature, dim=-1
                )
                
                distillation_loss = torch.nn.functional.kl_div(
                    student_probs, teacher_probs, reduction='batchmean'
                ) * (temperature ** 2)
                
                hard_loss = torch.nn.functional.cross_entropy(
                    student_output, batch['target']
                )
                
                total_loss = alpha * distillation_loss + (1 - alpha) * hard_loss
                
                total_loss.backward()
                optimizer.step()
        
        return self.student_model
    
    def feature_distillation(self, training_data, epochs=10):
        optimizer = torch.optim.Adam(self.student_model.parameters(), lr=1e-4)
        
        for epoch in range(epochs):
            for batch in training_data:
                optimizer.zero_grad()
                
                with torch.no_grad():
                    teacher_features = self.teacher_model.get_features(batch['input'])
                
                student_features = self.student_model.get_features(batch['input'])
                
                feature_loss = torch.nn.functional.mse_loss(
                    student_features, teacher_features
                )
                
                output = self.student_model(batch['input'])
                task_loss = torch.nn.functional.cross_entropy(output, batch['target'])
                
                total_loss = feature_loss + task_loss
                
                total_loss.backward()
                optimizer.step()
        
        return self.student_model
```

```python
class PerformanceMonitor:
    def __init__(self):
        self.metrics = defaultdict(list)
        self.monitoring = False
        self.monitor_thread = None
    
    def start_monitoring(self, interval=1):
        self.monitoring = True
        
        self.monitor_thread = threading.Thread(
            target=self._monitor_loop,
            args=(interval,)
        )
        self.monitor_thread.daemon = True
        self.monitor_thread.start()
    
    def stop_monitoring(self):
        self.monitoring = False
        
        if self.monitor_thread:
            self.monitor_thread.join()

        while self.monitoring:
            self._collect_metrics()
            time.sleep(interval)
    
    def _collect_metrics(self):
        timestamp = time.time()
        
        cpu_percent = psutil.cpu_percent(interval=None)
        self.metrics['cpu_percent'].append((timestamp, cpu_percent))
        
        memory = psutil.virtual_memory()
        self.metrics['memory_percent'].append((timestamp, memory.percent))
        self.metrics['memory_used'].append((timestamp, memory.used))
        
        try:
            gpus = GPUtil.getGPUs()
            for i, gpu in enumerate(gpus):
                self.metrics[f'gpu_{i}_load'].append((timestamp, gpu.load * 100))
                self.metrics[f'gpu_{i}_memory'].append((timestamp, gpu.memoryUsed))
        except:
            pass

        timestamp = time.time()
        
        self.metrics['request_latency'].append((timestamp, latency))
        self.metrics['tokens_generated'].append((timestamp, tokens_generated))
        self.metrics['tokens_per_second'].append(
            (timestamp, tokens_generated / latency if latency > 0 else 0)
        )
    
    def get_metrics(self):
        return dict(self.metrics)
    
    def get_summary(self):
        summary = {}
        
        for metric_name, values in self.metrics.items():
            if values:
                data = [v[1] for v in values]
                
                summary[metric_name] = {
                    'mean': sum(data) / len(data),
                    'min': min(data),
                    'max': max(data),
                    'count': len(data)
                }
        
        return summary
    
    def export_metrics(self, filename):
        metrics_data = {
            'metrics': dict(self.metrics),
            'summary': self.get_summary()
        }
        
        with open(filename, 'w') as f:
            json.dump(metrics_data, f, indent=2)

            new_replicas = max(self.current_replicas - 1, self.min_replicas)

            new_replicas = self.current_replicas
        
        self.current_replicas = new_replicas
        
        return new_replicas
```

```python
import threading
import queue
import time
import json
import logging
from typing import Dict, List
import hashlib

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class ProductionLLMService:
    def __init__(self, model_path, config):
        self.config = config
        
        self.model = torch.load(model_path)
        
        self.request_queue = queue.Queue(maxsize=config['max_queue_size'])
        self.response_cache = {}
        
        self.workers = []
        for i in range(config['num_workers']):
            worker = threading.Thread(target=self._worker_loop, args=(i,))
            worker.daemon = True
            worker.start()
            self.workers.append(worker)
        
        self.monitor = PerformanceMonitor()
        self.monitor.start_monitoring()
        
        self.auto_scaler = AutoScaler(
            min_replicas=config['min_replicas'],
            max_replicas=config['max_replicas']
        )
    
    def _worker_loop(self, worker_id):
        logger.info(f"Worker {worker_id} started")
        
        while True:
            try:
                request_id, prompt, params = self.request_queue.get(timeout=1)
                
                cache_key = self._get_cache_key(prompt, params)
                
                if cache_key in self.response_cache:
                    response = self.response_cache[cache_key]
                    logger.info(f"Worker {worker_id}: Cache hit for request {request_id}")
                else:
                    response = self._generate(prompt, params)
                    
                    self.response_cache[cache_key] = response
                    
                    if len(self.response_cache) > self.config['max_cache_size']:
                        self._evict_cache()
                    
                    logger.info(f"Worker {worker_id}: Generated response for request {request_id}")
                
                self.response_queues[request_id].put({
                    'status': 'success',
                    'response': response,
                    'worker_id': worker_id
                })
                
            except queue.Empty:
                continue
            except Exception as e:
                logger.error(f"Worker {worker_id}: Error - {str(e)}")
    
    def _generate(self, prompt, params):
        inputs = self.tokenizer(prompt, return_tensors="pt")
        
        start_time = time.time()
        
        with torch.no_grad():
            outputs = self.model.generate(
                inputs.input_ids,
                max_length=params.get('max_length', 100),
                temperature=params.get('temperature', 0.7),
                top_p=params.get('top_p', 0.9)
            )
        
        end_time = time.time()
        
        response = self.tokenizer.decode(outputs[0], skip_special_tokens=True)
        
        tokens_generated = outputs.shape[1] - inputs.input_ids.shape[1]
        
        self.monitor.record_request(
            hashlib.md5(prompt.encode()).hexdigest(),
            end_time - start_time,
            tokens_generated
        )
        
        return response
    
    def _get_cache_key(self, prompt, params):
        key_data = {
            'prompt': prompt,
            'params': {
                'max_length': params.get('max_length', 100),
                'temperature': params.get('temperature', 0.7),
                'top_p': params.get('top_p', 0.9)
            }
        }
        
        return hashlib.md5(json.dumps(key_data).encode()).hexdigest()
    
    def _evict_cache(self):
        keys_to_remove = list(self.response_cache.keys())[:self.config['cache_evict_count']]
        
        for key in keys_to_remove:
            del self.response_cache[key]
        
        logger.info(f"Evicted {len(keys_to_remove)} cache entries")
    
    def submit_request(self, prompt, params):
        if self.request_queue.full():
            return {'status': 'error', 'error': 'Queue full'}
        
        request_id = hashlib.md5(f"{prompt}{time.time()}".encode()).hexdigest()
        
        self.response_queues[request_id] = queue.Queue()
        
        self.request_queue.put((request_id, prompt, params))
        
        return {'status': 'queued', 'request_id': request_id}
    
    def get_response(self, request_id, timeout=30):
        try:
            response = self.response_queues[request_id].get(timeout=timeout)
            del self.response_queues[request_id]
            return response
        except queue.Empty:
            return {'status': 'timeout', 'error': 'Request timeout'}
    
    def get_status(self):
        return {
            'queue_size': self.request_queue.qsize(),
            'cache_size': len(self.response_cache),
            'workers': len(self.workers),
            'metrics': self.monitor.get_summary()
        }
```
