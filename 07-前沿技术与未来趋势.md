```
┌─────────────────────────────────┐
│  原理：验证推理结果             │
│                                 │

│        求面积"                  │
│                                 │
│  初步推理：                     │
│  这是直角三角形（3²+4²=5²）     │

│                                 │
│  验证步骤：                     │
│  1. 检查是否直角三角形          │

│                                 │
│  2. 检查面积计算                │
│     直角三角形面积 = (直角边1×直角边2)/2│

│                                 │
│  3. 使用海伦公式验证            │
│     s = (3+4+5)/2 = 6           │
│     面积 = √(6×3×2×1) = √36 = 6 ✓│
│                                 │
│  答案：6（验证通过）            │
│                                 │
│  效果：                         │
│  - 提高准确性                   │
│  - 减少错误                     │
```

### 💻 推理增强实现

```python
from typing import Dict, List, Tuple
import json

class ReasoningEnhancer:
    def __init__(self, llm_client):
        self.llm_client = llm_client
    
    def chain_of_thought(self, problem: str) -> Dict:
        prompt = f"""请使用思维链推理解决以下问题：

问题：{problem}

请逐步思考：
1. 首先，理解问题的核心
2. 然后，分析问题的关键要素
3. 接着，逐步推理得出结论
4. 最后，验证答案的正确性

请以以下格式输出：
思考1：[第一步思考]
思考2：[第二步思考]
思考3：[第三步思考]
...
答案：[最终答案]

推理过程："""
        
        response = self.llm_client.chat.completions.create(
            model="gpt-4",
            messages=[{"role": "user", "content": prompt}],
            temperature=0.5
        )
        
        reasoning_process = response.choices[0].message.content
        
        answer = self.extract_answer(reasoning_process)
        
        return {
            'problem': problem,
            'reasoning_process': reasoning_process,
            'answer': answer,
            'method': 'chain_of_thought'
        }
    
    def tree_of_thought(self, problem: str, num_paths: int = 3) -> Dict:
        prompt = f"""请使用思维树推理解决以下问题：

问题：{problem}

请探索{num_paths}条不同的推理路径，然后选择最佳路径。

请以以下格式输出：
路径1：
- 思考：[路径1的思考]
- 评估：[路径1的评估]

路径2：
- 思考：[路径2的思考]
- 评估：[路径2的评估]

路径3：
- 思考：[路径3的思考]
- 评估：[路径3的评估]

最佳路径：[选择的路径]
答案：[最终答案]

推理过程："""
        
        response = self.llm_client.chat.completions.create(
            model="gpt-4",
            messages=[{"role": "user", "content": prompt}],
            temperature=0.7
        )
        
        reasoning_process = response.choices[0].message.content
        
        answer = self.extract_answer(reasoning_process)
        
        return {
            'problem': problem,
            'reasoning_process': reasoning_process,
            'answer': answer,
            'method': 'tree_of_thought'
        }
    
    def graph_of_thought(self, problem: str) -> Dict:
        prompt = f"""请使用思维图推理解决以下问题：

问题：{problem}

请构建推理图，包括：
1. 识别问题的子问题
2. 分析子问题之间的关系
3. 逐步解决每个子问题
4. 融合所有结果得出最终答案

请以以下格式输出：
子问题1：[第一个子问题]
解决1：[子问题1的解决]

子问题2：[第二个子问题]
解决2：[子问题2的解决]

关系分析：[子问题之间的关系]
融合结果：[融合所有子问题的结果]
答案：[最终答案]

推理过程："""
        
        response = self.llm_client.chat.completions.create(
            model="gpt-4",
            messages=[{"role": "user", "content": prompt}],
            temperature=0.6
        )
        
        reasoning_process = response.choices[0].message.content
        
        answer = self.extract_answer(reasoning_process)
        
        return {
            'problem': problem,
            'reasoning_process': reasoning_process,
            'answer': answer,
            'method': 'graph_of_thought'
        }
    
    def self_consistency(self, problem: str, num_paths: int = 5) -> Dict:
        paths = []
        
        for i in range(num_paths):
            prompt = f"""请使用不同的方法解决以下问题：

问题：{problem}

请使用方法{i+1}解决，并给出推理过程。

推理过程{i+1}："""
            
            response = self.llm_client.chat.completions.create(
                model="gpt-4",
                messages=[{"role": "user", "content": prompt}],
                temperature=0.8
            )
            
            path_result = response.choices[0].message.content
            paths.append({
                'path_id': i+1,
                'reasoning': path_result,
                'answer': self.extract_answer(path_result)
            })
        
        aggregated_answer = self.aggregate_answers(paths)
        
        return {
            'problem': problem,
            'paths': paths,
            'aggregated_answer': aggregated_answer,
            'method': 'self_consistency'
        }
    
    def verification(self, problem: str, initial_answer: str) -> Dict:
        prompt = f"""请验证以下答案的正确性：

问题：{problem}
初步答案：{initial_answer}

请进行验证：
1. 检查推理过程是否正确
2. 检查计算是否准确
3. 检查逻辑是否合理
4. 如果发现错误，请修正

验证过程："""
        
        response = self.llm_client.chat.completions.create(
            model="gpt-4",
            messages=[{"role": "user", "content": prompt}],
            temperature=0.5
        )
        
        verification_process = response.choices[0].message.content
        
        verified_answer = self.extract_answer(verification_process)
        
        return {
            'problem': problem,
            'initial_answer': initial_answer,
            'verification_process': verification_process,
            'verified_answer': verified_answer,
            'method': 'verification'
        }
    
    def extract_answer(self, reasoning_process: str) -> str:
        lines = reasoning_process.split('\n')
        
        for line in lines:
            if '答案' in line or 'Answer' in line:
                return line.split(':')[1].strip() if ':' in line else line
        
        return reasoning_process
    
    def aggregate_answers(self, paths: List[Dict]) -> str:
        answers = [path['answer'] for path in paths]
        
        answer_counts = {}
        for answer in answers:
            answer_counts[answer] = answer_counts.get(answer, 0) + 1
        
        most_common_answer = max(answer_counts, key=answer_counts.get)
        
        return most_common_answer

class AdvancedReasoningSystem:
    def __init__(self, llm_client):
        self.llm_client = llm_client
        self.enhancer = ReasoningEnhancer(llm_client)
    
    def solve_complex_problem(self, problem: str) -> Dict:
        cot_result = self.enhancer.chain_of_thought(problem)
        
        verification_result = self.enhancer.verification(
            problem, cot_result['answer']
        )
        
        if verification_result['verified_answer'] != cot_result['answer']:
            consistency_result = self.enhancer.self_consistency(problem)
            
            final_answer = consistency_result['aggregated_answer']
        else:
            final_answer = cot_result['answer']
        
        return {
            'problem': problem,
            'cot_result': cot_result,
            'verification_result': verification_result,
            'final_answer': final_answer,
            'confidence': self.calculate_confidence(verification_result)
        }
    
    def calculate_confidence(self, verification_result: Dict) -> float:
        verification_text = verification_result['verification_process']
        
        if '正确' in verification_text or '验证通过' in verification_text:
            return 0.9
        elif '修正' in verification_text:
            return 0.7
        else:
            return 0.5

    print(f"问题：{cot_result['problem']}")
    print(f"\n推理过程：")
    print(cot_result['reasoning_process'])
    print(f"\n答案：{cot_result['answer']}")

if __name__ == "__main__":
    test_reasoning_enhancer()
```

### 💻 长上下文实现

```python
import torch
import torch.nn as nn
import math

class LongContextModel(nn.Module):
    def __init__(self, vocab_size, d_model, n_heads, n_layers, max_seq_len=100000):
        super().__init__()
        
        self.embedding = nn.Embedding(vocab_size, d_model)
        
        self.rope = RoPEPositionalEncoding(d_model, max_seq_len)
        
        self.layers = nn.ModuleList([
            LongContextLayer(d_model, n_heads, max_seq_len)
            for _ in range(n_layers)
        ])
        
        self.output = nn.Linear(d_model, vocab_size)
    
    def forward(self, x):
        x = self.embedding(x)
        
        x = self.rope(x)
        
        for layer in self.layers:
            x = layer(x)
        
        output = self.output(x)
        
        return output

class RoPEPositionalEncoding(nn.Module):
    def __init__(self, d_model, max_seq_len=100000, base=10000):
        super().__init__()
        
        self.d_model = d_model
        
        inv_freq = 1.0 / (base ** (torch.arange(0, d_model, 2).float() / d_model))
        self.register_buffer('inv_freq', inv_freq)
    
    def forward(self, x):
        batch_size, seq_len, d_model = x.shape
        
        positions = torch.arange(seq_len, device=x.device).unsqueeze(0)
        
        freqs = torch.einsum('bi,j->bij', positions.float(), self.inv_freq)
        
        emb = torch.cat([freqs, freqs], dim=-1)
        cos = emb.cos()
        sin = emb.sin()
        
        x1 = x[..., :d_model//2]
        x2 = x[..., d_model//2:]
        
        rotated_x1 = x1 * cos[..., :d_model//2] - x2 * sin[..., :d_model//2]
        rotated_x2 = x1 * sin[..., :d_model//2] + x2 * cos[..., :d_model//2]
        
        return torch.cat([rotated_x1, rotated_x2], dim=-1)

class LongContextLayer(nn.Module):
    def __init__(self, d_model, n_heads, max_seq_len):
        super().__init__()
        
        self.attention = FlashAttention(d_model, n_heads)
        self.ffn = nn.Sequential(
            nn.Linear(d_model, d_model * 4),
            nn.GELU(),
            nn.Linear(d_model * 4, d_model)
        )
        
        self.norm1 = nn.LayerNorm(d_model)
        self.norm2 = nn.LayerNorm(d_model)
    
    def forward(self, x):
        att_output = self.attention(x)
        x = self.norm1(x + att_output)
        
        ffn_output = self.ffn(x)
        x = self.norm2(x + ffn_output)
        
        return x

class FlashAttention(nn.Module):
    def __init__(self, d_model, n_heads, block_size=256):
        super().__init__()
        
        self.d_model = d_model
        self.n_heads = n_heads
        self.head_dim = d_model // n_heads
        self.block_size = block_size
        
        self.W_q = nn.Linear(d_model, d_model)
        self.W_k = nn.Linear(d_model, d_model)
        self.W_v = nn.Linear(d_model, d_model)
        self.W_o = nn.Linear(d_model, d_model)
    
    def forward(self, x):
        batch_size, seq_len, _ = x.shape
        
        Q = self.W_q(x)
        K = self.W_k(x)
        V = self.W_v(x)
        
        Q = Q.view(batch_size, seq_len, self.n_heads, self.head_dim).transpose(1, 2)
        K = K.view(batch_size, seq_len, self.n_heads, self.head_dim).transpose(1, 2)
        V = V.view(batch_size, seq_len, self.n_heads, self.head_dim).transpose(1, 2)
        
        output = self.flash_attention_forward(Q, K, V)
        
        output = output.transpose(1, 2).contiguous().view(batch_size, seq_len, self.d_model)
        
        output = self.W_o(output)
        
        return output
    
    def flash_attention_forward(self, Q, K, V):
        batch_size, seq_len, n_heads, head_dim = Q.shape
        
        output = torch.zeros_like(Q)
        
        for i in range(0, seq_len, self.block_size):
            Q_block = Q[:, i:i+self.block_size, :, :]
            
            block_output = torch.zeros_like(Q_block)
            block_max = torch.full((batch_size, self.block_size, n_heads, 1), float('-inf'))
            block_sum = torch.zeros((batch_size, self.block_size, n_heads, 1))
            
            for j in range(0, seq_len, self.block_size):
                K_block = K[:, j:j+self.block_size, :, :]
                V_block = V[:, j:j+self.block_size, :, :]
                
                scores = torch.matmul(Q_block, K_block.transpose(-2, -1))
                scores = scores / math.sqrt(head_dim)
                
                new_max = torch.maximum(block_max, scores.max(dim=-1, keepdim=True)[0])
                
                exp_scores = torch.exp(scores - new_max)
                
                correction = torch.exp(block_max - new_max)
                
                block_sum = correction * block_sum + exp_scores.sum(dim=-1, keepdim=True)
                
                block_output = correction * block_output + torch.matmul(exp_scores, V_block)
                
                block_max = new_max
            
            output[:, i:i+self.block_size, :, :] = block_output / block_sum
        
        return output
```

```python
import torch
import torch.nn as nn
from typing import Dict, List

class UnifiedMultimodalModel(nn.Module):
    def __init__(self, vocab_size, d_model, n_heads, n_layers):
        super().__init__()
        
        self.text_encoder = TextEncoder(vocab_size, d_model)
        self.image_encoder = ImageEncoder(d_model)
        self.audio_encoder = AudioEncoder(d_model)
        self.video_encoder = VideoEncoder(d_model)
        
        self.multimodal_transformer = nn.Transformer(
            d_model=d_model,
            nhead=n_heads,
            num_encoder_layers=n_layers,
            num_decoder_layers=n_layers
        )
        
        self.output = nn.Linear(d_model, vocab_size)
    
    def forward(self, inputs: Dict):
        encoded_inputs = []
        
        if 'text' in inputs:
            text_features = self.text_encoder(inputs['text'])
            encoded_inputs.append(text_features)
        
        if 'image' in inputs:
            image_features = self.image_encoder(inputs['image'])
            encoded_inputs.append(image_features)
        
        if 'audio' in inputs:
            audio_features = self.audio_encoder(inputs['audio'])
            encoded_inputs.append(audio_features)
        
        if 'video' in inputs:
            video_features = self.video_encoder(inputs['video'])
            encoded_inputs.append(video_features)
        
        combined_features = torch.cat(encoded_inputs, dim=1)
        
        transformer_output = self.multimodal_transformer(combined_features, combined_features)
        
        output = self.output(transformer_output)
        
        return output

class TextEncoder(nn.Module):
    def __init__(self, vocab_size, d_model):
        super().__init__()
        
        self.embedding = nn.Embedding(vocab_size, d_model)
    
    def forward(self, text_tokens):
        return self.embedding(text_tokens)

class ImageEncoder(nn.Module):
    def __init__(self, d_model):
        super().__init__()
        
        self.conv = nn.Sequential(
            nn.Conv2d(3, 64, kernel_size=7, stride=2, padding=3),
            nn.ReLU(),
            nn.MaxPool2d(2),
            nn.Conv2d(64, 128, kernel_size=3, stride=2, padding=1),
            nn.ReLU(),
            nn.MaxPool2d(2),
            nn.Conv2d(128, d_model, kernel_size=3, stride=2, padding=1),
            nn.ReLU(),
        )
        
        self.proj = nn.Linear(d_model, d_model)
    
    def forward(self, images):
        features = self.conv(images)
        
        features = features.flatten(2).transpose(1, 2)
        
        features = self.proj(features)
        
        return features

class AudioEncoder(nn.Module):
    def __init__(self, d_model):
        super().__init__()
        
        self.conv = nn.Sequential(
            nn.Conv1d(1, 64, kernel_size=7, stride=2, padding=3),
            nn.ReLU(),
            nn.MaxPool1d(2),
            nn.Conv1d(64, 128, kernel_size=3, stride=2, padding=1),
            nn.ReLU(),
            nn.MaxPool1d(2),
            nn.Conv1d(128, d_model, kernel_size=3, stride=2, padding=1),
            nn.ReLU(),
        )
        
        self.proj = nn.Linear(d_model, d_model)
    
    def forward(self, audio):
        features = self.conv(audio.unsqueeze(1))
        
        features = features.transpose(1, 2)
        
        features = self.proj(features)
        
        return features

class VideoEncoder(nn.Module):
    def __init__(self, d_model):
        super().__init__()
        
        self.image_encoder = ImageEncoder(d_model)
        
        self.temporal_encoder = nn.LSTM(d_model, d_model, batch_first=True)
    
    def forward(self, videos):
        batch_size, num_frames, channels, height, width = videos.shape
        
        frame_features = []
        for i in range(num_frames):
            frame = videos[:, i, :, :, :]
            frame_feature = self.image_encoder(frame)
            frame_features.append(frame_feature)
        
        frame_features = torch.stack(frame_features, dim=1)
        
        temporal_features, _ = self.temporal_encoder(frame_features)
        
        return temporal_features

        'image': torch.randn(batch_size, 3, 224, 224),
        'audio': torch.randn(batch_size, 16000),
        'video': torch.randn(batch_size, 10, 3, 224, 224)
    }
    
    output = model(inputs)
    
    print(f"输入模态：{list(inputs.keys())}")
    print(f"输出形状：{output.shape}")

if __name__ == "__main__":
    test_multimodal()
```

## 7.6 总结与展望

### 📝 本章核心要点

1. **推理增强**：让LLM真正"思考"
   - Chain-of-Thought：逐步推理
   - Tree-of-Thought：多路径探索
   - Graph-of-Thought：复杂关系建模
   - Self-Consistency：多路径聚合
   - Verification：结果验证

2. **长上下文突破**：突破序列长度限制
   - 密集注意力优化：Flash Attention、Ring Attention
   - 位置编码改进：RoPE外推、NTK-aware插值
   - 架构创新：稀疏注意力、状态空间模型

3. **多模态融合**：全模态交互
   - 统一表示：所有模态转换为统一token
   - 跨模态对齐：CLIP对齐、联合训练
   - 模态融合：早期融合、晚期融合、混合融合

4. **能力扩展**：超越基础生成
   - 工具使用：API调用、代码执行
   - 规划能力：任务分解、步骤规划
   - 学习能力：在线学习、知识更新
   - 创造能力：内容创作、设计创新
   - 协作能力：多Agent协作、人机协作

5. **未来趋势**：更智能、更高效、更普及
   - 模型规模：更大模型、更小模型、模型家族
   - 模型能力：更强推理、更好理解、更高创造
   - 模型效率：更快推理、更低成本、更少资源
   - 模型应用：更广领域、更深渗透、更好体验
   - 模型安全：更可控、更安全、更可信
